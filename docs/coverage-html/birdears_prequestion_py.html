1"""This module implements pre-questions' progressions. 

2 

3Pre questions are chord progressions or notes played before the question is 

4played, so to affirmate the sound of the question's key. 

5 

6For example a common cadence is chords I-IV-V-I from the diatonic scale, which 

7in a key of `C` is `CM-FM-GM-CM` and in a key of `A` is `AM-DM-EM-AM`. 

8 

9Pre-question methods should be decorated with `register_prequestion_method` 

10decorator, so that they will be registered as a valid pre-question method. 

11""" 

12

24def register_prequestion_method(f, *args, **kwargs): 

25 """Decorator for prequestion method functions. 

26 

27 Functions decorated with this decorator will be registered in the 

28 `PREQUESTION_METHODS` global dict. 

29 """ 

30 

31 @wraps(f) 

32 def decorator(*args, **kwargs): 

33 return f(*args, **kwargs) 

34 

35 PREQUESTION_METHODS.update({f.__name__: f}) 

36 

37 return decorator

40class PreQuestion: 

41 

42 def __init__(self, method, question): 

43 """This class implements methods for different types of pre-question 

44 progressions. 

45 

46 Args: 

47 method (str): The method used in the pre question. 

48 question (obj): Question object from which to generate the 

49 pre-question sequence. 

50 """ 

51 

52 self.METHOD = PREQUESTION_METHODS[method] 

53 self.question = question 

54 

55 def __call__(self, *args, **kwargs): 

56 """Calls the resolution method and pass arguments to it. 

57 

58 Returns a `birdears.Sequence` object with the pre-question generated by 

59 the method. 

60 """ 

61 return self.METHOD(question=self.question, *args, **kwargs)
