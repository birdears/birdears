21def center_text(text, sep=True, nl=0): 

22 """This function returns input text centered according to terminal columns. 

23 

24 Args: 

25 text (str): The string to be centered, it can have multiple lines. 

26 sep (bool): Add line separator after centered text (True) or 

27 not (False). 

28 nl (int): How many new lines to add after text. 

29 """ 

30 

31 linelist = list(text.splitlines()) 

32 

33 # gets the biggest line 

34 biggest_line_size = 0 

35 for line in linelist: 

36 line_lenght = len(line.expandtabs()) 

37 if line_lenght > biggest_line_size: 

38 biggest_line_size = line_lenght 

39 

40 columns = COLS 

41 offset = biggest_line_size / 2 

42 perfect_center = columns / 2 

43 padsize = int(perfect_center - offset) 

44 spacing = ' ' * padsize # space char 

45 

46 text = str() 

47 for line in linelist: 

48 text += (spacing + line + '\n') 

49 

50 divider = spacing + ('─' * int(biggest_line_size)) # unicode 0x2500 

51 text += divider if sep else '' 

52 

53 text += nl * '\n' 

54 

55 return text
